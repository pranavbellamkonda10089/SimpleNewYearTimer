    <script>
        window.onload = function() {
            // Precise target: Jan 1st 2026
            const target = new Date(2026, 0, 1, 0, 0, 0).getTime();

            function sync() {
                const now = Date.now();
                const diff = target - now;

                if (diff <= 0) {
                    document.querySelector('.glass-card').innerHTML = 
                        "<div class='main-headline' style='font-size:3.5rem'>2026</div><div class='wishing-text'>WELCOME TO THE FUTURE</div>";
                    return;
                }

                const d = Math.floor(diff / 86400000);
                const h = Math.floor((diff % 86400000) / 3600000);
                const m = Math.floor((diff % 3600000) / 60000);
                const s = Math.floor((diff % 60000) / 1000);

                document.getElementById('d').innerText = d;
                document.getElementById('h').innerText = String(h).padStart(2, '0');
                document.getElementById('m').innerText = String(m).padStart(2, '0');
                document.getElementById('s').innerText = String(s).padStart(2, '0');
            }

            sync();
            setInterval(sync, 1000);
        };
    </script>
